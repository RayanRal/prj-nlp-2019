# Query to get the list of first and last names of people born in England

PREFIX  dbo:  <http://dbpedia.org/ontology/>
PREFIX  dbpedia:  <http://dbpedia.org/resource/>

SELECT distinct ?firstname ?lastname WHERE {

?person dbo:birthPlace dbpedia:England .
?person foaf:givenName ?firstname .
?person foaf:surname ?lastname .

}
